```ruby
In Model:
class Comment < ActiveRecord::Base
  belongs_to :image
  belongs_to :user
  validates :comment, length: {minimum: 3}
end
```

```ruby
In Controller:
def create
  @image = Image.find(params[:image_id])
  @comment = @image.comments.new(comment_params)
  @comment.user_id = current_user.id
  if @comment.save
    redirect_to image_path(@image)
  else
    render 'new'
  end
end
```
